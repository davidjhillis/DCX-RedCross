JavaScript
// Initialize IGX CMS
const igx = require('@igx/cms-sdk');

const client = new igx.Client({
  apiKey: process.env.IGX_API_KEY,
  endpoint: 'https://api.igx.com/v1'
});

// Fetch content
async function getContent(id) {
  try {
    const content = await client.content.get(id);
    return content;
  } catch (error) {
    console.error('Failed to fetch content:', error);
    throw error;
  }
}

export { getContent };
Python
from igx import Client

# Initialize client
client = Client(
    api_key="your_api_key_here",
    endpoint="https://api.igx.com/v1"
)

# Create new content
def create_content(title, body):
    """Create new content item"""
    response = client.content.create({
        "title": title,
        "body": body,
        "status": "draft"
    })
    return response

# Example usage
content = create_content("My Title", "Content body")
print(f"Created: {content.id}")

Use the igx.Client() method to initialize the SDK with your API credentials.
